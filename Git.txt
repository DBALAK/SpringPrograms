1) git init -- create .git folder to maintain metadata info
   git add . (pwd contents)
2) git commit -- track updated or new file/folder 
	git commit -m "first commit"
   git remote add origin https://github.com/DBALAK/SpringPrograms.git

5) git clone - eg. git clone https://github.com/DBALAK/SpringPrograms.git
6) git status - status of file in satging area.
7) git stash -- to put in temporary directory.
